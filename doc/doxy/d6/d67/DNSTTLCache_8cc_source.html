44 int DNSTTLCache::put_into_cache(std::shared_ptr<DNSRecord> record) {
45  // calculate the hash, check if it's already in the cache.
46 
47  if (!record->rdata && record->rtype == DNS_TYPE_VALUE_SRV) {
48  throw cRuntimeError("Retrieved invalid record to put into cache, with empty data");
49  }
50 
51  std::string type = std::string(getTypeStringForValue(record->rtype));
52  std::string _class = std::string(getClassStringForValue(record->rclass));
53 
54  // create hash:
55  std::string hash = record->rname + std::string(":") + type
56  + std::string(":") + _class;
57 
58  // check if it is in the cache
59  if (!is_in_cache(hash)) {
60  if (DNSCache::getCacheSize() > DNSCache::getMaxRecords()) {
61  evict();
62  }
63 
64  std::list<std::shared_ptr<DNSTimeRecord>> list;
65  std::shared_ptr<DNSTimeRecord> time_record(new DNSTimeRecord());
66 
67  time_record->record = record;
68  time_record->hash = hash;
69  time_record->rcv_time = simTime();
70 
71  std::string stime = std::to_string(record->ttl) + std::string("s");
72  simtime_t tv = simTime() + STR_SIMTIME(stime.c_str());
73  time_record->expiry = time_record->rcv_time + tv;
74 
75  //g_printf("DEBUG MSG: New hash entering CACHE --- [%s], for record->rname=%s\n", hash, record->rname);
76  list.push_back(time_record);
77  cache[hash] = list;
78  dnsRecordPriorityCache.insert(time_record);
79  setCacheSize(getCacheSize() + 1);
80  } else {
81  // check if the data is still the same
82  std::list<std::shared_ptr<DNSTimeRecord>> from_cache = cache[hash];
83 
84  int is_already_in_cache = 0;
85  for (auto it = from_cache.begin(); it != from_cache.end(); ++it) {
86  std::shared_ptr<DNSTimeRecord> record_from_cache = *it;
87 
88  if (ODnsExtension::recordDataEqual(record,
89  record_from_cache->record)) {
90  is_already_in_cache = 1;
91  // update expiry by new data
92  break;
93  }
94  }
95 
96  if (!is_already_in_cache) {
97  // create a new time record
98  std::shared_ptr<DNSTimeRecord> time_record(new DNSTimeRecord());
99  time_record->record = record;
100  time_record->hash = hash;
101  time_record->rcv_time = simTime();
102 
103  std::string stime = std::to_string(record->ttl) + std::string("s");
104  simtime_t tv = simTime() + STR_SIMTIME(stime.c_str());
105  time_record->expiry = time_record->rcv_time + tv;
106 
107  // replace the entry in the database
108  from_cache = cache[hash];
109  from_cache.push_back(time_record);
110  cache[hash] = from_cache;
111 
112  dnsRecordPriorityCache.insert(time_record);
113  }
114  }
115 
116  return 1;
117 
118 }

137 int DNSTTLCache::halfTTL(std::shared_ptr<ODnsExtension::DNSRecord> r) {
138  // get the time record list for the hash from the cache
139  std::string type = std::string(getTypeStringForValue(r->rtype));
140  std::string _class = std::string(getClassStringForValue(r->rclass));
141 
142  // create hash:
143  std::string hash = r->rname + std::string(":") + type + std::string(":")
144  + _class;
145  std::list<std::shared_ptr<DNSTimeRecord>> tr_list = cache[hash];
146  for (auto it = tr_list.begin(); it != tr_list.end(); ++it) {
147  std::shared_ptr<DNSTimeRecord> tr = *it;
148  std::shared_ptr<DNSRecord> record = tr->record;
149 
150  if (ODnsExtension::recordDataEqual(r, record)) {
151  std::string stime = std::to_string(r->ttl) + std::string("s");
152  simtime_t ttl_to_sim = STR_SIMTIME(stime.c_str());
153  simtime_t curr = simTime();
154  return (tr->expiry - curr).inUnit(-3) < (ttl_to_sim / 2).inUnit(-3);
155  }
156  }
157 
158  return 0;
159 }

161 std::list<std::shared_ptr<DNSRecord>> DNSTTLCache::cleanup() {
162  std::list<std::shared_ptr<DNSRecord>> returnlist;
163  for (auto iterator = dnsRecordPriorityCache.begin();
164  iterator != dnsRecordPriorityCache.end(); ++iterator) {
165  std::shared_ptr<DNSTimeRecord> r = *iterator;
166  if (r->expiry < simTime()) {
167  // remove from caches
168  dnsRecordPriorityCache.erase(iterator);
169  std::shared_ptr<DNSRecord> removed_record = remove_from_cache(
170  r->hash, r->record);
171  returnlist.push_back(removed_record);
172  } else {
173  break; // we're finished cleaning up.
174  }
175  }
176 
177  return returnlist;
178 }

180 std::list<std::shared_ptr<DNSRecord>> DNSTTLCache::remove_from_cache(
181  std::string hash) {
182  std::list<std::shared_ptr<DNSTimeRecord>> from_cache = cache[hash];
183  std::list<std::shared_ptr<DNSRecord>> return_list;
184  if (from_cache.size() > 0) {
185  cache.erase(hash);
186  setCacheSize(getCacheSize() - 1);
187 
188  // remove all time records from the priority queue ..
189  for (auto it = from_cache.begin(); it != from_cache.end(); ++it) {
190  return_list.push_back((*it)->record);
191  remove_time_record(*it);
192  }
193 
194  return return_list;
195  }
196 
197  return return_list;
198 }

200 std::shared_ptr<DNSRecord> DNSTTLCache::remove_from_cache(std::string hash,
201  std::shared_ptr<DNSRecord> r) {
202  std::list<std::shared_ptr<DNSTimeRecord>> from_cache = cache[hash];
203 
204  for (auto it = from_cache.begin(); it != from_cache.end(); ++it) {
205  std::shared_ptr<DNSTimeRecord> tr = *it;
206 
207  if (ODnsExtension::recordDataEqual(tr->record, r)) {
208  // we have the record
209  from_cache.erase(it);
210  // replace the list in the table
211  cache[hash] = from_cache;
212  // free the time record, return the record
213  std::shared_ptr<DNSRecord> return_record = tr->record;
214  remove_time_record(tr);
215  return return_record;
216  }
217  }
218 
219  // this means we did not find the record, return NULL
220  return NULL;
221 }

224  std::set<std::shared_ptr<DNSTimeRecord>,
225  ODnsExtension::DNSTimeRecordComparator>::iterator it;
226 
227  for (auto it = dnsRecordPriorityCache.begin();

236 std::list<std::shared_ptr<DNSRecord>> DNSTTLCache::evict() {
237  if (getCacheSize() == 0) {
238  std::list<std::shared_ptr<DNSRecord>> empty;
239  return empty;
240  }
241 
242  // evict top element...

244 
245  return remove_from_cache(top->hash);
