47 void MDNSQueryScheduler::elapse(ODnsExtension::TimeEvent* e,
48  std::shared_ptr<void> data) {
49  std::shared_ptr<ODnsExtension::MDNSQueryJob> qj = std::static_pointer_cast
50  < MDNSQueryJob > (data);
51 
52  if (qj->done) {
53  remove_job(qj); // remove the job from history as it is done already
54  return;
55  }
56 
57  int is_private = 0;
58  if (hasPrivacy) {
59  // check whether the record is of private nature
60  std::string service_type = ODnsExtension::extract_stype(qj->key->name);
61  if (private_service_table->find(service_type)
62  != private_service_table->end()) {
63  std::shared_ptr<ODnsExtension::PrivateMDNSService> psrv =
64  (*private_service_table)[service_type];
65  is_private = psrv->is_private;
66  }
67  }
68 
69  int packetSize = 12; // initial header size
70 
71  std::list<std::shared_ptr<DNSQuestion>> qlist;
72  std::list<std::shared_ptr<DNSRecord>> anlist, nslist, arlist;
73  int qdcount = 0, ancount = 0, nscount = 0, arcount = 0;
74 
75  int success = append_question(qj->key, &qlist, &anlist, &packetSize,
76  &qdcount, &ancount, is_private);
77  done(qj);
78 
79  // now try to append more questions if we didn't already exceed the packet size.
80  std::list<std::shared_ptr<MDNSQueryJob>> done_records;
81  if (!is_private) { // only if we don't have a private query job
82  for (auto job : jobs) {
83  // take another query job from the list..
84  if(!success) break;
85 
86  int _private_job = 0;
87  std::string service_type = ODnsExtension::extract_stype(
88  job->key->name);
89 
90  // check whether this service is private, do not append it if it is
91  if (hasPrivacy
92  && private_service_table->find(service_type)
93  != private_service_table->end()) {
94  std::shared_ptr<ODnsExtension::PrivateMDNSService> psrv =
95  (*private_service_table)[service_type];
96  _private_job = psrv->is_private;
97  // reschedule
98  timeEventSet->updateTimeEvent(job->e,
99  simTime() + STR_SIMTIME("20ms"));
100  }
101 
102  if (!_private_job) {
103  success = append_question(job->key, &qlist, &anlist,
104  &packetSize, &qdcount, &ancount, 0);
105  }
106  if (!_private_job) {
107  done_records.push_back(job);
108  }
109  }
110 
111  for(auto job : done_records) done(job);
112  }
113 
114  if (preparePacketAndSend(qlist, anlist, nslist, arlist, qdcount, ancount,
115  nscount, arcount, packetSize, !success, is_private)) {
116  // success, delegate?
117  } else {
118  // some error message?
119  }
120 
121 }

123 int MDNSQueryScheduler::preparePacketAndSend(
124  std::list<std::shared_ptr<DNSQuestion>> qlist,
125  std::list<std::shared_ptr<DNSRecord>> anlist,
126  std::list<std::shared_ptr<DNSRecord>> nslist,
127  std::list<std::shared_ptr<DNSRecord>> arlist, int qdcount, int ancount,
128  int nscount, int arcount, int packetSize, int TC, int is_private) {
129 
130  std::string msgname;
131  if (!is_private)
132  msgname = "MDNS_query#";
133  else
134  msgname = "PRIVATE_query#";
135 
136  msgname += std::to_string(id_count);
137 
138  DNSPacket* p = ODnsExtension::createNQuery(msgname, qdcount, ancount,
139  nscount, arcount, id_count++, 0);
140 
141  int i = 0;
142  // append questions
143  for (auto it : qlist) {
144  ODnsExtension::appendQuestion(p, it, i);
145  i++;
146  }
147 
148  // append answers if available
149  if (ancount > 0) {
150  i = 0;
151  for (auto it : anlist) {
152  ODnsExtension::appendAnswer(p, it, i);
153  i++;
154  }
155  }
156 
157  // append auth if available
158  if (nscount > 0) {
159  i = 0;
160  for (auto it : nslist) {
161  ODnsExtension::appendAuthority(p, it, i);
162  i++;
163  }
164  }
165 
166  // append add if available
167  if (arcount > 0) {
168  i = 0;
169  for (auto it : arlist) {
170  ODnsExtension::appendAdditional(p, it, i);
171  i++;
172  }
173  }
174 
175  // packet fully initialized, send it via multicast
176  p->setByteLength(packetSize);
177  if (!is_private) {
178  const char* dstr = "i=msg/bcast,red";
179  p->setDisplayString(dstr);
180  outSock->sendTo(p, multicast_address, MDNS_PORT);
181  } else {
182  const char* dstr = "i=msg/packet,green";
183  p->setDisplayString(dstr);
184  std::string service_type = ODnsExtension::extract_stype(
185  p->getQuestions(0).qname);
186  std::shared_ptr<ODnsExtension::PrivateMDNSService> psrv =
187  (*private_service_table)[service_type];
188  // go through the offered_to list
189  for (auto it : psrv->offered_to) {
190  std::string key = it;
191  std::shared_ptr<ODnsExtension::FriendData> fdata =
192  (*friend_data_table)[key];
193  if (fdata && fdata->online) {
194  // send per TCP to the privacy socket on the given port
195  privacySock->sendTo(p->dup(), fdata->address, fdata->port);
196  }
197  }
198 
199  delete p;
200  }
201 
202  // packet is out, we're finished
203  return 1;
204 }

206 int MDNSQueryScheduler::append_question(std::shared_ptr<MDNSKey> key,
207  std::list<std::shared_ptr<DNSQuestion>>* qlist,
208  std::list<std::shared_ptr<DNSRecord>>* anlist, int *packetSize,
209  int* qdcount, int* ancount, int is_private) {
210  std::list<std::shared_ptr<DNSRecord>> knownAnswers; // append known answers from the cache in here
211 
212  std::shared_ptr<ODnsExtension::DNSQuestion> q;
213  q = createQuestionFromKey(key);
214 
215  int qsize = sizeof(key->name) + 4; // name length + 4 bytes for type and class
216 
217  if (*packetSize + qsize > MAX_MDNS_PACKET_SIZE) {
218  return 0;
219  }
220 
221  *packetSize += qsize;
222 
223  (*qdcount)++; // this throws a warning, but we actually want to increase the referenced value ..
224  (*qlist).push_back(q);
225 
226  // append known answers for this query
227  if (!is_private) { // do not append KA if the query job is private
228  knownAnswers = append_cache_entries(key, knownAnswers);
229 
230  // try to append known answers, as long as max size is not exceeded
231  for (auto it = knownAnswers.begin(); it != knownAnswers.end(); ++it) {
232  std::shared_ptr<DNSRecord> record = *it;
233 
234  // calculate size
235  int size = 10 + sizeof(record->rname) + record->rdlength;
236 
237  if (*packetSize + size > MAX_MDNS_PACKET_SIZE) {
238  return 0;
239  }
240 
241  *packetSize += size;
242 
243  // append record to answer list
244  (*ancount)++;
245  (*anlist).push_back(record);
246  }
247  }
248  // all answers were appended, return success
249  return 1;
250 }

252 std::list<std::shared_ptr<DNSRecord>> MDNSQueryScheduler::append_cache_entries(
253  std::shared_ptr<MDNSKey> key,
254  std::list<std::shared_ptr<DNSRecord>> list) {
255  std::string hash = key->name + std::string(":")
256  + std::string(getTypeStringForValue(key->type)) + std::string(":")
257  + std::string(getClassStringForValue(key->_class));
258  std::list<std::shared_ptr<DNSRecord>> from_cache = cache->get_from_cache(
259  hash);
260  std::shared_ptr<DNSRecord> record;
261 
262  for (auto it = from_cache.begin(); it != from_cache.end(); ++it) {
263  record = *it;
264 
265  // try to append known answer if halfTTL not outlived..
266  if (cache->halfTTL(record)) {
267  // everything is fine, we can append the answer..
268  // FIXME: create a copy of the record, this way it is
269  // too unsafe.
270  list.push_back(record);
271  }
272  }
273 
274  return list;

307 void MDNSQueryScheduler::post(std::shared_ptr<ODnsExtension::MDNSKey> key,
308  int immediately) {
309  std::shared_ptr<MDNSQueryJob> qj;
310  simtime_t tv;
311 
312  if ((qj = find_history(key)))
313  return;
314 
315  if (!immediately) {
316  int defer = intrand(100) + 20;
317  // create simtime value from random deferral value
318  std::string stime = std::to_string(defer) + std::string("ms");
319  tv = simTime() + STR_SIMTIME(stime.c_str());
320 
321  } else {
322  tv = simTime();
323  }
324 
325  // update time if this question is a duplicate
326  if ((qj = find_job(key))) {
327  if (tv < qj->delivery) {
328  qj->delivery = tv;
329  timeEventSet->updateTimeEvent(qj->e, tv);
330  }
331  } else {
332  // create new job..
333  qj = new_job(key);
334  qj->delivery = tv;
335 
336  ODnsExtension::TimeEvent* e = new ODnsExtension::TimeEvent(this);
337  e->setData(qj);
338  e->setExpiry(tv);
339  e->setLastRun(0);
340  e->setCallback(ODnsExtension::MDNSQueryScheduler::elapseCallback);
341 
342  qj->e = e;
343  timeEventSet->addTimeEvent(e);
344 
345  }
346 
347  std::shared_ptr<SimTime> tv_ptr(new SimTime(tv));
348 
349  callback(tv_ptr, resolver);
350 
351 }

353 void MDNSQueryScheduler::done(std::shared_ptr<ODnsExtension::MDNSQueryJob> qj) {
354  qj->done = 1;
355 
356  auto it = std::find(jobs.begin(), jobs.end(), qj);
357  if (it != jobs.end()) {
358  jobs.erase(it);
359  }
360  if (std::find(history.begin(), history.end(), qj) == history.end())
361  return;
362 
363  history.push_back(qj);
364 
365  simtime_t now = simTime();
366  qj->delivery = now;
367 
368  // update the time event
369 
370  // add random deferral value between 20 and 120
371  int defer = intrand(100) + 20;
372  // create simtime value from random deferral value
373  std::string stime = std::to_string(defer) + std::string("ms");
374  simtime_t tv = STR_SIMTIME(stime.c_str());
375 
376  timeEventSet->updateTimeEvent(qj->e, now + tv);
377 }

379 void MDNSQueryScheduler::check_dup(
380  std::shared_ptr<ODnsExtension::MDNSKey> key) {
381  std::shared_ptr<MDNSQueryJob> qj;
382 
383  if ((qj = find_job(key))) {
384  // found a matching upcoming job, we don't need
385  // to perform it anymore, since another node
386  // queried for it.
387 
388  done(qj);
389  return;
390  }
391 
392  // add random deferral value between 20 and 120
393  int defer = intrand(100) + 20;
394  // create simtime value from random deferral value
395  std::string stime = std::to_string(defer) + std::string("ms");
396  simtime_t tv = simTime() + STR_SIMTIME(stime.c_str());
397 
398  if ((qj = find_history(key))) {
399  // just update the time for the existing job
400  qj->delivery = tv;
401  timeEventSet->updateTimeEvent(qj->e, tv);
402  }
403 
404 }

422 void MDNSQueryScheduler::remove_job(
423  std::shared_ptr<ODnsExtension::MDNSQueryJob> qj) {
424  timeEventSet->removeTimeEvent(qj->e);
425 
426  if (find_job(qj->key)) {
427  auto it = std::find(jobs.begin(), jobs.end(), qj);
428  if (it != jobs.end()) {
429  jobs.erase(it);
430  }
431 
432  return;
433  } else if (find_history(qj->key)) {
434  auto it = std::find(history.begin(), history.end(), qj);
435  if (it != history.end()) {
436  history.erase(it);
437  }
438 
439  return;
440  }
441 
442  // no ref found? i.e. just delete ...
443  qj.reset();
444 }
