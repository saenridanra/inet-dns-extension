130  unsigned short rd, unsigned short ra, unsigned short rcode) {

132  // Set id and options in header ..
133  r->setId(id);
134  r->setQdcount(qdcount);
135  r->setNumQuestions(qdcount);
136  r->setAncount(ancount);
137  r->setNumAnswers(ancount);
138  r->setNscount(nscount);
139  r->setNumAuthorities(nscount);
140  r->setArcount(arcount);
141  r->setNumAdditional(arcount);
142 
143  unsigned short options = 0;
144  DNS_HEADER_SET_QR(options, 1);
145  // Opcode based on query..
146  DNS_HEADER_SET_OPCODE(options, opcode);
147  // Authoritative? Application should know ..
148  DNS_HEADER_SET_AA(options, AA);
149  // Recursion desired ..
150  DNS_HEADER_SET_RD(options, rd);
151  DNS_HEADER_SET_RA(options, ra);
152 
153  DNS_HEADER_SET_RCODE(options, rcode);
154 
155  // Append answers separately in another method
156 
157  r->setOptions(options);
158 
159  return r;
160 }

206 std::shared_ptr<Response> resolveResponse(cPacket* response) {
207  DNSPacket* v = dynamic_cast<DNSPacket*>(response);
208 
209  if (v == 0) // Bad .. not a DNSPacket!
210  {
211  throw E_NOT_A_DNSPACKET
212  ;
213  }
214 
215  if (DNS_HEADER_QR(v->getOptions()) != 1) {
216  throw E_WRONG_QR
217  ;
218  }
219 
220  // Migrate Answers
221  DNSRecord answers[v->getNumAnswers()];
222  for (short i = 0; i < v->getAncount(); i++) {
223  answers[i] = v->getAnswers(i);
224  }
225 
226  // Migrate Authoritative Records
227  DNSRecord authoritative[v->getNumAuthorities()];
228  for (short i = 0; i < v->getNscount(); i++) {
229  authoritative[i] = v->getAuthorities(i);
230  }
231 
232  // Migrate Additional Records
233  DNSRecord additional[v->getNumAdditional()];
234  for (short i = 0; i < v->getArcount(); i++) {
235  additional[i] = v->getAdditional(i);
236  }
237 
238  std::shared_ptr<Response> r(
239  new Response(v->getId(), v->getOptions(), v->getQdcount(), 0, 0, 0,
240  answers, authoritative, additional));
241 
242  return r;
243 }

646 int recordDataEqual(std::shared_ptr<DNSRecord> r1,
647  std::shared_ptr<DNSRecord> r2) {
648  if (r1->rname != r2->rname
649  && (r1->rtype != r2->rtype || r1->rclass != r2->rclass))
650  return 0;
651 
652  switch (r1->rtype) {
653  case DNS_TYPE_VALUE_SRV: {
654  std::shared_ptr<SRVData> s1 = std::static_pointer_cast<SRVData> (r1->rdata);
655  std::shared_ptr<SRVData> s2 = std::static_pointer_cast<SRVData> (r2->rdata);
656 
657  if (s1->name == s2->name && s1->port == s2->port
658  && s1->priority == s2->priority && s1->proto == s2->proto
659  && s1->service == s2->service && s1->target == s2->target
660  && s1->weight == s2->weight)
661  return 1;
662  break;
663  }
664  default: // compare str value
665  return r1->strdata == r2->strdata;
666  }
667 
668  return 0;
669
