79 ODnsExtension::TimeEvent* TimeEventSet::getTimeEventIfDue(){
80  if(timeEventSet.empty()) return NULL;
81 
82  simtime_t now = simTime();
83  std::set<ODnsExtension::TimeEvent*, ODnsExtension::TimeEventComparator>::iterator it = timeEventSet.begin();
84  ODnsExtension::TimeEvent* top = *it;
85  if(top->getExpiry() <= now){
86  // if another timevent is scheduled then it will
87  // be readded by the scheduler accordingly so
88  // it is at it's designated position. For now we
89  // can remove it from the set.
90  timeEventSet.erase(it);
91  return top;
92  }
93 
94  return NULL;
95 }
