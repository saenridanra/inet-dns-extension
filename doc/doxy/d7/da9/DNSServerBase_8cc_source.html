43 void DNSServerBase::handleMessage(cMessage *msg)
44 {
45  int isDNS = 0;
46  int isQR = 0;
47  std::shared_ptr<ODnsExtension::Query> query;
48  DNSPacket* response;
49 
50  // Check if we received a query
51  if (msg->arrivedOn("udpIn"))
52  {
53  if ((isDNS = ODnsExtension::isDNSpacket((cPacket*) msg)))
54  {
55  if ((isQR = ODnsExtension::isQueryOrResponse((cPacket*) msg)) == 0)
56  {
57 
58  query = ODnsExtension::resolveQuery((cPacket*) msg);
59  receivedQueries++;
60 
61  cPacket *pk = check_and_cast<cPacket *>(msg);
62  UDPDataIndication *ctrl = check_and_cast<UDPDataIndication *>(pk->getControlInfo());
63  IPvXAddress srcAddress = ctrl->getSrcAddr();
64  query->src_address = srcAddress.str();
65  response = handleQuery(query);
66 
67  if (response == NULL)
68  { // only happens if recursive resolving was initiated
69  delete msg;
70  return;
71  }
72 
73  // and send the response to the source address
74  sendResponse(response, srcAddress);
75  }
76  else
77  {
78  // Just got a response, lets see if its an answer fitting one of
79  // the queries we need to resolved.
80  response = handleRecursion((DNSPacket*) msg);
81 
82  if (response != NULL)
83  {
84  // this was the final answer, i.e.
85  // get the original packet and the src addr
86  int id = ((DNSPacket*) msg)->getId();
87  std::shared_ptr<ODnsExtension::CachedQuery> cq = get_query_from_cache(id);
88 
89  IPvXAddress addr = IPvXAddressResolver().resolve(cq->query->src_address.c_str());
90 
91  // free cached query data
92  remove_query_from_cache(id, cq);
93 
94  // we're not an authority, set it here.
95  sendResponse(response, addr);
96  }
97 
98  }
99  }
100 
101  }
102 
103  delete msg;
104 
105 }

107 DNSPacket* DNSServerBase::handleRecursion(DNSPacket* packet)
108 {
109  // first check if we have a query id that belongs to this packet
110  // and the answer relates to the query
111 
112  DNSPacket* response;
113 
114  if (!queryCache.count(packet->getId()))
115  {
116  return NULL; // we do not have a query that belongs to this key
117  }
118 
119  std::shared_ptr<ODnsExtension::CachedQuery> cq = queryCache[packet->getId()];
120  std::shared_ptr<ODnsExtension::Query> original_query = cq->query;
121 
122  // first check, see if there are actually answers
123 
124  if (DNS_HEADER_AA(packet->getOptions()) && packet->getAncount() > 0)
125  {
126  // we have what we looked for, return
127  std::string msg_name = std::string("dns_response#") + std::to_string(original_query->id);
128  response = ODnsExtension::createResponse(msg_name, 1, packet->getAncount(), packet->getNscount(),
129  packet->getArcount(), original_query->id, DNS_HEADER_OPCODE(original_query->options), 0,
130  DNS_HEADER_RD(original_query->options), 1, 0);
131 
132  short i;
133  for (i = 0; i < cq->query->qdcount; i++)
134  {
135  ODnsExtension::appendQuestion(response, ODnsExtension::copyDnsQuestion(&cq->query->questions[i]), i);
136  }
137 
138  std::string bubble_popup = "";
139  for (i = 0; i < packet->getAncount(); i++)
140  {
141 
142  // store the response in the cache
143  if (responseCache)
144  {
145  // check if the record is not an A or AAAA record
146  if (packet->getAnswers(i).rtype != DNS_TYPE_VALUE_A
147  && packet->getAnswers(i).rtype != DNS_TYPE_VALUE_AAAA)
148  {
149  //create a copy and put it into the cache
150  std::shared_ptr<ODnsExtension::DNSRecord> r = ODnsExtension::copyDnsRecord(&(packet->getAnswers(i)));
151 
152  // put the record into the cache
153  bubble_popup.append("New cache entry:\n");
154  bubble_popup.append(r->rname.c_str());
155  bubble_popup.append(":");
156  bubble_popup.append(ODnsExtension::getTypeStringForValue(r->rtype));
157  bubble_popup.append(":");
158  bubble_popup.append(ODnsExtension::getClassStringForValue(r->rclass));
159  bubble_popup.append("\nData: ");
160  bubble_popup.append(r->strdata.c_str());
161  bubble_popup.append("\n---------\n");
162  responseCache->put_into_cache(r);
163  }
164  }
165 
166  ODnsExtension::appendAnswer(response, ODnsExtension::copyDnsRecord(&packet->getAnswers(i)), i);
167  }
168 
169  if (bubble_popup != "")
170  {
171  EV << bubble_popup.c_str();
172  this->getParentModule()->bubble(bubble_popup.c_str());
173  }
174 
175  if (responseCache && original_query->questions[0].qname == packet->getQuestions(0).qname)
176  {
177  // we have a mismatch in the queries, this means we followed a CNAME chain
178  // and used the end of chain to query the server, hence we need to append
179  // the CNAME chain
180 
181  std::string cnhash = original_query->questions[0].qname + ":" + DNS_TYPE_STR_CNAME + ":" +
182  DNS_CLASS_STR_IN;
183  std::list<std::string> hashes = responseCache->get_matching_hashes(cnhash);
184 
185  int num_hashes = hashes.size();
186  // reset size of answers to ancount + hashes length
187  response->setNumAnswers(response->getAncount() + num_hashes);
188  response->setAncount(response->getAncount() + num_hashes);
189 
190  int pos = packet->getAncount();
191  for(auto it = hashes.begin(); it != hashes.end(); it++)
192  {
193  // use the hash to get the corresponding entry
194  std::string tmp = (std::string) *it;
195  std::list<std::shared_ptr<DNSRecord>> records = responseCache->get_from_cache(tmp);
196 
197  if (!records.empty())
198  break;
199 
200  // list should not be greater one otherwise there is a collision
201  if (records.size() > 1)
202  {
203  responseCache->remove_from_cache(tmp);
204  break;
205  }
206 
207  // only one record, extract data into tmp
208  if ((*(records.begin()))->rtype == DNS_TYPE_VALUE_CNAME)
209  {
210  // append record to the section
211  ODnsExtension::appendAnswer(response, ODnsExtension::copyDnsRecord(*(records.begin())),
212  pos);
213  pos++;
214  }
215  }
216  }
217 
218  for (i = 0; i < packet->getNscount(); i++)
219  {
220  ODnsExtension::appendAuthority(response, ODnsExtension::copyDnsRecord(&packet->getAuthorities(i)), i);
221  }
222  for (i = 0; i < packet->getArcount(); i++)
223  {
224  ODnsExtension::appendAdditional(response, ODnsExtension::copyDnsRecord(&packet->getAdditional(i)), i);
225  }
226 
227  return response;
228  }
229  else if (DNS_HEADER_AA(packet->getOptions()) && packet->getAncount() == 0)
230  {
231  // return the entry not found response
232  std::string msg_name = "dns_response#" + std::to_string(original_query->id);
233 
234  response = ODnsExtension::createResponse(msg_name, 1, 0, 0, 0, original_query->id,
235  DNS_HEADER_OPCODE(original_query->options), 1, DNS_HEADER_RD(original_query->options), 1, 3);
236 
237  for (int i = 0; i < cq->query->qdcount; i++)
238  {
239  ODnsExtension::appendQuestion(response, ODnsExtension::copyDnsQuestion(&cq->query->questions[i]), i);
240  }
241 
242  return response; // return the response with no entry found..
243  }
244  else if (packet->getNscount() > 0 && packet->getArcount() > 0 && !DNS_HEADER_AA(packet->getOptions()))
245  {
246  // we have an answer for a query
247  // pick one at random and delegate the question
248 
249  int p = intrand(packet->getNscount());
250  std::shared_ptr<DNSRecord> r = ODnsExtension::copyDnsRecord(&packet->getAdditional(p));
251 
252  // query the name server for our original query
253  std::string msg_name = "dns_query#" + std::to_string(cq->internal_id) + std::string("--recursive");
254  DNSPacket *query = ODnsExtension::createQuery(msg_name, packet->getQuestions(0).qname, DNS_CLASS_IN,
255  packet->getQuestions(0).qtype, cq->internal_id, 1);
256 
257  // Resolve the ip address for the record
258  IPvXAddress address = IPvXAddressResolver().resolve(r->strdata.c_str());
259 
260  if (!address.isUnspecified())
261  sendResponse(query, address);
262 
263  return NULL; // since this packet is fine we pass it upwards
264  }
265  else if (packet->getNscount() > 0 && !DNS_HEADER_AA(packet->getOptions()))
266  {
267  // TODO: no ar record, we need to start at the beginning with this reference..
268  return NULL;
269  }
270  else
271  {
272  // something went wrong, return a server failure query
273  std::string msg_name = "dns_response#" + std::to_string(original_query->id);
274  response = ODnsExtension::createResponse(msg_name, 1, 0, 0, 0, original_query->id,
275  DNS_HEADER_OPCODE(original_query->options), 0, DNS_HEADER_RD(original_query->options), 1, 2);
276 
277  return response; // return the response with no entry found..
278  }
279 
280  return NULL;
281 }

313 {
