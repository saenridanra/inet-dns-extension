41 DNSPacket* DNSLocalCache::handleQuery(std::shared_ptr<ODnsExtension::Query> query)
42 {
43  DNSPacket* response;
44  ODnsExtension::DNSQuestion q;
45  int id, opcode, rd, ra, an_records = 0,ns_records = 0, ar_records = 0, stop_cache_lookup = 0, rec_query_created = 0;
46  std::list<std::shared_ptr<DNSRecord>> answer_list, ns_list, ar_list;
47  const char* __class;
48  const char* type;
49  std::string msg_name, namehash;
50 
51  if (query->qdcount > 1)
52  {
53  response = DNSServerBase::unsupportedOperation(query);
54  return response;
55  }
56 
57  // initializes options
58  id = query->id;
59 
60  opcode = DNS_HEADER_OPCODE(query->options);
61  // recursion desired?
62  rd = DNS_HEADER_RD(query->options);
63 
64  // Go through all questions and generate answers.
65  q = query->questions[0];
66 
67  // generate msg name
68  msg_name = std::string("dns_response#") + std::to_string(response_count++);
69 
70  // init class
71  switch (q.qclass)
72  {
73  case DNS_CLASS_IN:
74  __class = DNS_CLASS_STR_IN;
75  break;
76  case DNS_CLASS_CH:
77  case DNS_CLASS_CS:
78  case DNS_CLASS_HS:
79  case DNS_CLASS_ANY:
80  default:
81  response = DNSServerBase::unsupportedOperation(query);
82  return response;
83  }
84 
85  // init type
86  // at the moment we only support A, AAAA for local
87  // cache for usage in combination with the echo server
88  switch (q.qtype)
89  {
90  case DNS_TYPE_VALUE_A:
91  type = DNS_TYPE_STR_A;
92  break;
93  case DNS_TYPE_VALUE_AAAA:
94  type = DNS_TYPE_STR_AAAA;
95  break;
96  case DNS_TYPE_VALUE_CNAME:
97  case DNS_TYPE_VALUE_HINFO:
98  case DNS_TYPE_VALUE_MINFO:
99  case DNS_TYPE_VALUE_MX:
100  case DNS_TYPE_VALUE_NS:
101  case DNS_TYPE_VALUE_NULL:
102  case DNS_TYPE_VALUE_PTR:
103  case DNS_TYPE_VALUE_SOA:
104  case DNS_TYPE_VALUE_TXT:
105  case DNS_TYPE_VALUE_SRV:
106  case DNS_TYPE_VALUE_ANY:
107  case DNS_TYPE_VALUE_AXFR:
108  response = DNSServerBase::unsupportedOperation(query);
109  return response;
110  }
111 
112  // first check the cache
113  if((q.qtype == DNS_TYPE_VALUE_A || q.qtype == DNS_TYPE_VALUE_AAAA) && responseCache){
114  if (!rd || !recursion_available){
115  // response with not found err, as recursion is not desired or not available
116  // and we don't cache A records directly
117  response = ODnsExtension::createResponse(msg_name, 1, an_records, ns_records, ar_records, id, opcode, 0,
118  rd, ra, 0);
119  return response;
120  }
121 
122  // we know we don't store A and AAAA records in the cache
123  // i.e., check if we have a corresponding CNAME mapping in the
124  // cache
125  std::string cnhash = q.qname + std::string(":") + std::string(DNS_TYPE_STR_CNAME) + std::string(":") + std::string(__class);
126  std::list<std::string> hashes = responseCache->get_matching_hashes(cnhash);
127 
128  // walk through the hashes and initiate recursive queries
129  // for this simulation we can assume that we will only get one hash
130  // for which there are no double entries in the cache
131 
132  for(auto it = hashes.begin(); it != hashes.end() && !stop_cache_lookup && !rec_query_created; ++it){
133  // use the hash to get the corresponding entry
134  std::string tmp = *it;
135  // check if there is and entry in the cache, if so
136  // we should follow it before querying
137  std::list<std::shared_ptr<DNSRecord>> records;
138  while(responseCache->is_in_cache(tmp)){
139  // get the list of records from the cache
140  records = responseCache->get_from_cache(tmp);
141  // this list should not be greater than one, if it is
142  // the cache may have been polluted, delete all entries
143  // and do the normal iterative query
144 
145  if(records.size() > 1){
146  responseCache->remove_from_cache(tmp);
147  stop_cache_lookup = 1;
148  break;
149  }
150 
151  // only one record, extract data into tmp
152  if((*(records.begin()))->rtype == DNS_TYPE_VALUE_CNAME){
153  tmp = (*(records.begin()))->strdata + std::string(":") + std::string(DNS_TYPE_STR_CNAME) + std::string(":") + std::string(DNS_CLASS_STR_IN);
154  }
155  else // end of chain but not a CNAME
156  break;
157  }
158 
159  // if the flag is not set, we can use the record to perform our recursive query
160  if(!stop_cache_lookup && rootServers.size() > 0){
161  // the record is stored in *records
162  std::shared_ptr<DNSRecord> end_of_chain_record = *(records.begin());
163  // use the rdata in the record to create a recursive query
164  int id = DNSServerBase::getIdAndInc();
165  DNSServerBase::store_in_query_cache(id, query);
166  msg_name = std::string("dns_query#") + std::to_string(id) + std::string("--recursive");
167 
168  int p = intrand(rootServers.size());
169  DNSPacket *root_q = ODnsExtension::createQuery(msg_name, end_of_chain_record->strdata, DNS_CLASS_IN,
170  query->questions[0].qtype, id, 1);
171 
172  DNSServerBase::sendResponse(root_q, rootServers[p]);
173 
174  rec_query_created = 1;
175  }
176  }
177 
178  if(!stop_cache_lookup && rec_query_created) // there were no problems, recursive query was generated
179  return NULL;
180 
181  }
182 
183  // if we get here, cache lookup was unsuccessful
184  if (recursion_available && rootServers.size() > 0)
185  {
186  if (rd)
187  {
188  int id = DNSServerBase::getIdAndInc();
189  DNSServerBase::store_in_query_cache(id, query);
190  msg_name = std::string("dns_query#") + std::to_string(id) + std::string("--recursive");
191 
192  // do the initial query towards a root server
193  // pick at random
194  int p = intrand(rootServers.size());
195  DNSPacket *root_q = ODnsExtension::createQuery(msg_name, query->questions[0].qname, DNS_CLASS_IN,
196  query->questions[0].qtype, id, 1);
197 
198  DNSServerBase::sendResponse(root_q, rootServers[p]);
199 
200  return NULL;
201 
202  }
203  else
204  {
205  // response with not found err
206  response = ODnsExtension::createResponse(msg_name, 1, an_records, ns_records, ar_records, id, opcode, 0,
207  rd, ra, 0);
208 
209  // set question
210  response->setQuestions(0, q);
211  return response;
212  }
213  }
214  else
215  {
216  //TODO; Iterative answer.
217  response = DNSServerBase::unsupportedOperation(query);
218  return response;
219  }
220 
221 }
