35 void DNSZoneConfig::initialize(std::string config_file)
36 {
37  std::string line;
38  std::string lastsuffix;
39  std::shared_ptr<zone_entry> e;
40  std::string namehash;
41 
42  std::fstream conf(config_file, std::ios::in);
43  while (getline(conf, line, '\n'))
44  {
45 
46 #ifdef DEBUG_ENABLED
47  std::cout << "Processing:" << line << std::endl;
48 #endif
49 
50  if (line.empty() || line[0] == ';')
51  {
52 
53 #ifdef DEBUG_ENABLED
54  std::cout << "Skipping" << std::endl;
55 #endif
56 
57  continue;
58  }
59 
60  // use a tokenizer to interpret the line
61  std::vector<std::string> tokens = cStringTokenizer(line.c_str()).asVector();
62 
63  if (tokens.size() == 0)
64  {
65  continue; // only white spaces in this line ..
66  }
67 
68 #ifdef DEBUG_ENABLED
69  std::cout << "Number of tokens: " << tokens.size() << std::endl;
70 
71  for(uint32_t i = 0; i < tokens.size(); i++)
72  {
73  std::cout << "tokens[" << i << "] = " << tokens[i] << "; " << std::endl;
74  }
75 
76 #endif
77 
78  switch (state)
79  {
80  case states::VARS:
81  if (line[0] == '$')
82  {
83  // This is a variable and the first two tokens are relevant, i.e.
84  // $VAR VALUE
85  if (tokens[0] == "$TTL")
86  TTL = std::stoi(tokens[1].substr(1));
87  else if (tokens[0] == "$ORIGIN")
88  {
89  origin = std::string(tokens[1]);
90  }
91 
92  }
93  else if (tokens[2] == "SOA")
94  {
95  state = states::SOA;
96 #ifdef DEBUG_ENABLED
97  std::cout << "State set to \"SOA\"\n";
98 #endif
99  // init SOA first line
100  lastsuffix = std::string(tokens[0]);
101  zone_soa->mname = std::string(tokens[3]);
102  zone_soa->rname = std::string(tokens[4]);
103 
104  }
105  else
106  {
107  // malformed?
108  }
109  break;
110  case states::SOA:
111  if (tokens[0] != ")")
112  {
113  // init value
114  if (tokens[2] == "sn")
115  {
116  zone_soa->serial = std::stoi(tokens[0]);
117  }
118  else if (tokens[2] == "ref")
119  {
120  zone_soa->refresh = std::stoi(tokens[0]);
121  }
122  else if (tokens[2] == "ret")
123  {
124  zone_soa->retry = std::stoi(tokens[0]);
125  }
126  else if (tokens[2] == "ex")
127  {
128  zone_soa->expire = std::stoi(tokens[0]);
129  }
130  else if (tokens[2] == "nx")
131  {
132  zone_soa->minimum = std::stoi(tokens[0]);
133  }
134  }
135  else
136  {
137  state = states::ENTRY;
138  }
139 
140  break;
141  case states::ENTRY:
142  if (tokens[0] == ";")
143  break;
144 
145  // REMARK: Ignoring TTL in entries, or more precise,
146  // not allowing it right now, so leave it out of the zone file
147  // we use the last known suffix
148  e = std::shared_ptr<zone_entry>(new zone_entry());
149 
150  if (tokens[0] == "IN" || tokens[0] == "CS" || tokens[0] == "HS" || tokens[0] == "CH"
151  || tokens[0] == "*")
152  {
153  if (lastsuffix == "@")
154  {
155  e->domain = std::string(origin);
156  }
157  else
158  {
159  e->domain = std::string(lastsuffix);
160  }
161  e->__class = std::string(tokens[0]);
162  e->type = std::string(tokens[1]);
163 
164  e->data = std::string(tokens[2]);
165  }
166  else
167  {
168  // should start with the suffix
169  if (tokens[0] == "@")
170  {
171  e->domain = std::string(origin);
172  }
173  else if (ODnsExtension::stdstr_has_suffix(tokens[0], std::string(".")))
174  {
175  e->domain = std::string(tokens[0]);
176  }
177  else
178  {
179  e->domain = std::string(tokens[0]) + "." + std::string(origin);
180  }
181 
182  lastsuffix = std::string(tokens[0]);
183  e->__class = std::string(tokens[1]);
184  e->type = std::string(tokens[2]);
185  e->data = std::string(tokens[3]);
186 
187  }
188 
189  if (!ODnsExtension::stdstr_has_suffix(e->domain, std::string("."))
190  && !ODnsExtension::stdstr_has_suffix(e->domain, origin))
191  {
192  namehash = e->domain + "." + origin + ":" + e->type + ":" + e->__class;
193  }
194  else
195  {
196  namehash = e->domain + ":" + e->type + ":" + e->__class;
197  }
198 
199  if(zone_catalog.find(namehash) != zone_catalog.end())
200  {
201  std::list<std::shared_ptr<zone_entry>> list = zone_catalog[namehash];
202  list.push_back(e);
203  zone_catalog[namehash] = list;
204  }
205  else
206  {
207  std::list<std::shared_ptr<zone_entry>> first_element;
208  first_element.push_back(e);
209  zone_catalog[namehash] = first_element;
210  }
211 
212 #ifdef DEBUG_ENABLED
213  std::cout << "Inserted" << namehash << "into hashtable\n";
214 #endif
215 
216  break;
217 
218  default:
219  break;
220  }
221  }
222 
223 #ifdef DEBUG_ENABLED
224  std::cout << "Fully initialized zone configuration.\n";
225 #endif
226 }
