116 void MDNSResolver::handleMessage(cMessage *msg) {
117  if (msg->isSelfMessage()) {
118  if (msg->getKind() == MDNS_KIND_TIMER) {
119  elapsedTimeCheck();
120  return;
121  }
122  } else {
123  // check which kind it is
124  if (msg->getKind() == MDNS_KIND_INTERNAL_QUERY) {
125  // this is a message from a module utilizing the mdns resolver
126  // and wanting to perform a query..
127  delete msg;
128  return;
129  } else if (msg->getKind() == MDNS_KIND_INTERNAL_PUBLISH) {
130  // a higher layer application wants to publish a service
131  delete msg;
132  return;
133  } else if (msg->getKind() == MDNS_KIND_INTERNAL_REVOKE) {
134  // a higher layer application wants to revoke a service offered
135  delete msg;
136  return;
137  } else if (msg->getKind() == UDP_I_DATA) {
138  DNSPacket* p = check_and_cast<DNSPacket*>(msg);
139  UDPDataIndication *ctrl = check_and_cast<UDPDataIndication *>(
140  p->getControlInfo());
141  IPvXAddress srcAddress = ctrl->getSrcAddr();
142 
143  if (srcAddress.get4() != hostaddress.get4()) {
144  if (ODnsExtension::isQuery(p)) {
145  handleQuery(p);
146  } else if (ODnsExtension::isResponse(p)) {
147  handleResponse(p);
148  }
149  }
150 
151  delete msg;
152  return;
153  }
154 
155  }

210 void MDNSResolver::handleQuery(DNSPacket* p) {
211  // go through the ns section of the query
212  std::list<std::shared_ptr<DNSRecord>> record_list;
213 
214  // perform a cache cleanup, every entry that has passed
215  // it's TTL was not successfully updated
216  cache->cleanup();
217 
218  // go through the question section, find out which answers to respond with
219  for (int i = 0; i < p->getQdcount(); i++) {
220  ODnsExtension::DNSQuestion question = p->getQuestions(i);
221  std::shared_ptr<ODnsExtension::MDNSKey> key = ODnsExtension::mdns_key_new(
222  question.qname, question.qtype, question.qclass);
223  // allow suppression if there are no answers for this query
224  // and the tc flag is not set
225  if (p->getAncount() == 0 && !DNS_HEADER_TC(p->getOptions()))
226  queryScheduler->check_dup(key);
227 
228  // prepare matching responses
229  std::list<std::shared_ptr<DNSRecord>> announced_records =
230  announcer->get_announced_services();
231 
232  for (auto it = announced_records.begin(); it != announced_records.end(); ++it) {
233  // check if the record matches the key, if so append it to the answer list
234  std::shared_ptr<DNSRecord> r = *it;
235  std::shared_ptr<ODnsExtension::MDNSKey> record_key = ODnsExtension::mdns_key_new(
236  r->rname, r->rtype, r->rclass);
237 
238  // ANY Question
239  if (key->type == DNS_TYPE_VALUE_ANY) {
240  // only compare name and class
241  if (ODnsExtension::compareMDNSKeyANY(key, record_key)) {
242  record_list.push_back(r);
243  }
244  } // Normal Question
245  else {
246  if (ODnsExtension::compareMDNSKey(key, record_key)) {
247  record_list.push_back(r);
248  }
249  }
250 
251  // If the question did non include ANY or CNAME, we did not check for CNAMEs just yet
252  if (key->type != DNS_TYPE_VALUE_CNAME
253  && key->type != DNS_TYPE_VALUE_ANY) {
254  std::shared_ptr<ODnsExtension::MDNSKey> cname_key = ODnsExtension::mdns_key_new(
255  question.qname, DNS_TYPE_VALUE_CNAME, question.qclass);
256  if (ODnsExtension::compareMDNSKey(cname_key, record_key)) {
257  record_list.push_back(r);
258  }
259 
260  // free the cname key
261  ODnsExtension::mdns_key_free(cname_key);
262  }
263 
264  ODnsExtension::mdns_key_free(record_key);
265  }
266 
267  // free the question key
268  ODnsExtension::mdns_key_free(key);
269  }
270 
271  // go through the answer section and perform KAS
272  IPvXAddress* querier = &(check_and_cast<UDPDataIndication *>(
273  p->getControlInfo()))->getSrcAddr();
274  for (int i = 0; i < p->getAncount(); i++) {
275  std::shared_ptr<DNSRecord> answer = ODnsExtension::copyDnsRecord(&p->getAnswers(i));
276  responseScheduler->suppress(answer, 0, querier, 0);
277  // remove records that we may have appended to the list
278  // since the records pointers are different we have to go through the list ..
279  for (auto it = record_list.begin(); it != record_list.end(); ++it) {
280  std::shared_ptr<DNSRecord> in_record = *it;
281  if (ODnsExtension::recordDataEqual(in_record, answer)) {
282  // remove the record from the list ..
283  record_list.erase(it++);
284  }
285  }
286  }
287 
288  // now check probes and check whether they collide
289  for (int i = 0; i < p->getNscount(); i++) {
290  std::shared_ptr<DNSRecord> ns_record = ODnsExtension::copyDnsRecord(&p->getAuthorities(i));
291  announcer->check_conflict(ns_record); // check whether we have a problem
292  }
293 
294  // we're finished, now let's post the responses
295  for (auto it = record_list.begin(); it != record_list.end(); ++it) {
296  responseScheduler->post(*it, 0, querier,
297  p->getNscount() > 0);
298  }
299 }

301 void MDNSResolver::handleResponse(DNSPacket* p) {
302  // go through the answer list of the packet
303  std::string bubble_popup = "";
304  for (int i = 0; i < p->getAncount(); i++) {
305  // check if the record conflicts, if not put it into the cache
306  std::shared_ptr<DNSRecord> r = ODnsExtension::copyDnsRecord(&p->getAnswers(i));
307  std::string type = std::string(ODnsExtension::getTypeStringForValue(r->rtype));
308  std::string _class = std::string(ODnsExtension::getClassStringForValue(r->rclass));
309  // create hash:
310  std::string hash = r->rname + std::string(":") + type + std::string(":") + _class;
311  if (r->rtype != DNS_TYPE_VALUE_ANY) {
312  if (!announcer->check_conflict(r) && !cache->is_in_cache(hash)) {
313  // put the record into the cache
314  bubble_popup.append("New cache entry:\n");
315  bubble_popup.append(r->rname);
316  bubble_popup.append(":");
317  bubble_popup.append(
318  ODnsExtension::getTypeStringForValue(r->rtype));
319  bubble_popup.append(":");
320  bubble_popup.append(
321  ODnsExtension::getClassStringForValue(r->rclass));
322  bubble_popup.append("\nData: ");
323  bubble_popup.append(r->strdata.c_str());
324  bubble_popup.append("\n---------\n");
325 
326  cache->put_into_cache(ODnsExtension::copyDnsRecord(r));
327 
328  // look if this record belongs to a friend announcing a privacy service
329  if (hasPrivacy && r->rtype == DNS_TYPE_VALUE_SRV) {
330  // check probe whether it is a friend, then we can set the status to online
331  if (instance_name_table->find(r->rname) != instance_name_table->end()) {
332  // the hash table contains the user in question
333  std::shared_ptr<ODnsExtension::FriendData> fdata = (*instance_name_table)[r->rname];
334  // set to online and last_informed
335  if ((fdata->last_informed
336  < simTime() - STR_SIMTIME("30s")
337  || fdata->last_informed <= STR_SIMTIME("0s"))
338  && !ODnsExtension::isGoodbye(r)) {
339  IPvXAddress querier =
340  check_and_cast<UDPDataIndication *>(
341  p->getControlInfo())->getSrcAddr();
342 
343  fdata->address = IPvXAddress(querier.get4());
344 
345  // copy address, set it in fdata
346  fdata->address = querier;
347  fdata->last_informed = simTime();
348  fdata->online = 1;
349 
350  std::shared_ptr<DNSRecord> record(new DNSRecord());
351  record->rname = own_instance_name + std::string("._privacy._tcp.local");
352  record->rtype = DNS_TYPE_VALUE_SRV;
353  record->rclass = DNS_CLASS_IN;
354 
355  std::shared_ptr<ODnsExtension::SRVData> srv(new ODnsExtension::SRVData());
356  srv->name = record->rname;
357  srv->service = std::string("._privacy._tcp.local");
358  srv->target = hostname + std::string(".local");
359  srv->port = 9977; // TODO: remove hard coded port ..
360  srv->priority = 0;
361  srv->weight = 0;
362  srv->proto = "_tcp";
363 
364  record->rdata = srv;
365  record->rdlength = 6 + srv->name.length() + srv->service.length() + srv->target.length();
366 
367  std::cout << "[" << hostname << "]: Friend " << fdata->pdata->friend_id << "came online\n";
368  responseScheduler->post(record, 0, NULL, 0); // post our own response into the scheduler
369  // it will be checked and sent via the privacy socket
370  } else if (ODnsExtension::isGoodbye(r)) {
371  // user went offline
372  fdata->online = 0;
373  }
374  }
375  }
376 
377  responseScheduler->check_dup(r, 0);
378  }
379  }
380  }
381 
382  if (bubble_popup != "") {
383  EV << bubble_popup.c_str();
384  this->getParentModule()->bubble(bubble_popup.c_str());
385  }

401 void MDNSResolver::initializeServiceFile(std::string file) {
402  std::string line;
403  std::fstream service_file(file, std::ios::in);
404  int error = 0;
405 
406  std::shared_ptr<ODnsExtension::MDNSService> service(new ODnsExtension::MDNSService);
407 
408  while (getline(service_file, line, '\n')) {
409 
410  if (line.empty() || line[0] == ';') {
411 
412  continue;
413  }
414 
415  // use a tokenizer to interpret the line
416  std::vector<std::string> tokens =
417  cStringTokenizer(line.c_str(), "=").asVector();
418 
419  if (tokens.size() != 2) {
420  continue; // we have exactly two tokens vor our key value pairs
421  }
422 
423  if (tokens[0] == "service_type") {
424  service->service_type = tokens[1];
425  } else if (tokens[0] == "instance_name") {
426  service->name = tokens[1];
427  } else if (tokens[0] == "txt_record") {
428  service->txtrecords.push_back(tokens[1].c_str());
429  } else if (tokens[0] == "port") {
430  char* tail;
431  service->port = strtol(tokens[1].c_str(), &tail, 10);
432  } else {
433  error = 1;
434  cRuntimeError("Malformed service file %s", file.c_str());
435  }
436 
437  }
438 
439  if (!error) {
440  services.push_back(service);
441  }

445 void MDNSResolver::initializePrivateServices() {
446  private_service_table = new std::unordered_map<std::string, std::shared_ptr<ODnsExtension::PrivateMDNSService>>();
447  friend_data_table = new std::unordered_map<std::string, std::shared_ptr<ODnsExtension::FriendData>>();
448  instance_name_table = new std::unordered_map<std::string, std::shared_ptr<ODnsExtension::FriendData>>();
449 
450  probeScheduler->setPrivacyData(private_service_table, friend_data_table,
451  instance_name_table, &privacySock);
452  queryScheduler->setPrivacyData(private_service_table, friend_data_table,
453  instance_name_table, &privacySock);
454  responseScheduler->setPrivacyData(private_service_table, friend_data_table,
455  instance_name_table, &privacySock);
456 
457  // first read pairing data param and initialize
458  const char* pairing_data = par("pairing_data").stringValue();
459  // use tokenizer to separate by ; , which separates entries
460  cStringTokenizer tokenizer(pairing_data, ";");
461  const char *token;
462  const char *inner_token;
463 
464  while (tokenizer.hasMoreTokens()) {
465  // initialize service file
466  token = tokenizer.nextToken();
467  // separate by , with new tokenizer
468  cStringTokenizer inner_tokenizer(token, ",");
469  int pos = 0;
470  std::string crypto_key;
471  std::string friend_id;
472  std::string privacy_service_instance_name;
473 
474  while (inner_tokenizer.hasMoreTokens()) {
475  inner_token = inner_tokenizer.nextToken();
476  // the format of pairing data is fixed, first the friend id
477  // then the privacy_instance_name and then the crypto key
478  switch (pos) {
479  case 0:
480  friend_id = std::string(inner_token);
481  break;
482  case 1:
483  privacy_service_instance_name = std::string(inner_token);
484  break;
485  case 2:
486  crypto_key = std::string(inner_token);
487  break;
488  default:
489  break;
490  }
491 
492  pos++;
493  }
494 
495  // create a pairing data and friend data object, insert it into the hash table
496  std::shared_ptr<ODnsExtension::PairingData> pdata = ODnsExtension::pairing_data_new(
497  crypto_key, friend_id, privacy_service_instance_name);
498  std::shared_ptr<ODnsExtension::FriendData> fdata = ODnsExtension::friend_data_new(pdata,
499  DEFAULT_PRIVACY_SOCKET_PORT);
500 
501  std::cout << "[" << hostname << "]: Adding friend " << friend_id << " with instance name " << privacy_service_instance_name << "\n";
502  (*friend_data_table)[friend_id] = fdata;
503  (*instance_name_table)[privacy_service_instance_name] =
504  fdata;
505  }
506 
507  // now initialize private service files
508 
509  const char* private_service_files =
510  par("privacy_service_files").stringValue();
511  const char* file;
512  cStringTokenizer tokenizer2(private_service_files);
513  while (tokenizer2.hasMoreTokens()) {
514  // init file by file:
515  file = tokenizer2.nextToken();
516  std::string line;
517  std::fstream private_service_file(file, std::ios::in);
518 
519  std::list<std::string> offered_to;
520  std::list<std::string> offered_by;
521  std::string stype;
522  int is_private;
523 
524  // go through the lines of the file
525  while (getline(private_service_file, line, '\n')) {
526  // parse line by '=' delimiter
527  cStringTokenizer inner_tokenizer(line.c_str(), "=");
528  // two tokens, label and value
529  std::vector<std::string> tokens = inner_tokenizer.asVector();
530 
531  if (tokens.size() != 2)
532  cRuntimeError("Error initializing private service file %s",
533  file);
534 
535  if (tokens[0] == "type") {
536  stype = tokens[1];
537  } else if (tokens[0] == "is_private") {
538  if (tokens[1] == "0")
539  is_private = 0;
540  else if (tokens[1] == "1")
541  is_private = 1;
542  else
543  cRuntimeError(
544  "Error initializing private service file %s. Wrong value for is_private",
545  file);
546  } else if (tokens[0] == "offered_to") {
547  // parse list, comma separated
548  std::vector<std::string> offers = cStringTokenizer(
549  tokens[1].c_str(), ",").asVector();
550  // put the offers into the list
551  for (unsigned int i = 0; i < offers.size(); i++)
552  offered_to.push_back(offers[i]);
553  } else if (tokens[0] == "offered_by") {
554  // parse list, comma separated
555  std::vector<std::string> offers = cStringTokenizer(
556  tokens[1].c_str(), ",").asVector();
557  // put the offers into the list
558  for (unsigned int i = 0; i < offers.size(); i++)
559  offered_by.push_back(offers[i]);
560  } else {
561  cRuntimeError(
562  "Unrecognized line parsing private service file %s.",
563  file);
564  }
565 
566  }
567 
568  // now populate the private service object, store it in the hash table
569  std::shared_ptr<ODnsExtension::PrivateMDNSService> psrv =
570  ODnsExtension::private_service_new(stype, is_private);
571  psrv->offered_to = offered_to;
572  psrv->offered_by = offered_by;
573  (*private_service_table)[stype] = psrv;
574 
575  }
