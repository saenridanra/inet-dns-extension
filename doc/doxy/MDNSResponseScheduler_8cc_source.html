142 void MDNSResponseScheduler::remove_job(std::shared_ptr<MDNSResponseJob> rj) {
143  timeEventSet->removeTimeEvent(rj->e);
144 
145  if (rj->done) {
146  auto it = std::find(history.begin(), history.end(), rj);
147  if (it != history.end()) {
148  it = history.erase(it);
149  freeDnsRecord(rj->r);
150  }
151  return;
152  } else if (rj->suppressed) {
153  auto it = std::find(suppressed.begin(), suppressed.end(), rj);
154  if (it != suppressed.end()) {
155  it = suppressed.erase(it);
156  freeDnsRecord(rj->r);
157  }
158  return;
159  } else {
160  auto it = std::find(jobs.begin(), jobs.end(), rj);
161  if (it != jobs.end()) {
162  it = jobs.erase(it);
163  freeDnsRecord(rj->r);
164  }
165  return;
166  }
167 
168 // no ref found? i.e. just delete ...
169  freeDnsRecord(rj->r);
170  rj.reset();

295 int MDNSResponseScheduler::appendFromCache(std::string hash,
296  std::list<std::shared_ptr<DNSRecord>> *anlist, int* packetSize,
297  int* ancount) {
298  std::list<std::shared_ptr<DNSRecord>> cache_entries;
299  std::shared_ptr<DNSRecord> from_cache;
300  cache_entries = auth_cache->get_from_cache(hash);
301 
302  // for each cache entry, call append record..
303  for (auto from_cache : cache_entries) {
304  int size = 10 + sizeof(from_cache->rname) + from_cache->rdlength;
305 
306  if (from_cache->rtype == DNS_TYPE_VALUE_SRV)
307  size += 6; // since we do not have WEIGHT, PRIO and PORT in the SRV
308  // record, we add these 6 bytes.
309 
310  if (*packetSize + size > MAX_MDNS_PACKET_SIZE) {
311  return 0;
312  }
313 
314  *packetSize += size;
315 
316  appendRecord(from_cache, anlist, packetSize, ancount);
317  (*ancount)++;
318  }
319  return 1;
320 }

341 int MDNSResponseScheduler::preparePacketAndSend(
342  std::list<std::shared_ptr<DNSRecord>> anlist, int ancount,
343  int packetSize, int is_private) {
344 
345  std::string msgname;
346  if (!is_private)
347  msgname = "MDNS_response#";
348  else
349  msgname = "PRIVATE_response#";
350 
351  msgname += std::to_string(id_count);
352 
353  DNSPacket* p = createResponse(msgname, 0, ancount, 0, 0, id_count, 0, 1, 0,
354  0, 0);
355 
356 // append answers if available
357  int i = 0;
358 // append questions
359  if (ancount > 0) {
360  for (auto it = anlist.begin(); it != anlist.end(); ++it) {
361  appendAnswer(p, *it, i);
362  i++;
363  }
364  }
365 
366  p->setByteLength(packetSize);
367  if (!is_private) {
368  const char* dstr = "i=msg/bcast,red";
369  p->setDisplayString(dstr);
370  outSock->sendTo(p, multicast_address, MDNS_PORT);
371  } else {
372  const char* dstr = "i=msg/packet,green";
373  p->setDisplayString(dstr);
374  std::string service_type = extract_stype(p->getAnswers(0).rname);
375  std::shared_ptr<PrivateMDNSService> psrv =
376  (*private_service_table)[service_type];
377  // go through the offered_to list
378 
379  for (auto key : psrv->offered_to) {
380  std::shared_ptr<FriendData> fdata = (*friend_data_table)[key];
381  if (fdata && fdata->online) {
382  // send per TCP to the privacy socket on the given port
383  privacySock->sendTo(p->dup(), fdata->address, fdata->port);
384  }
385  }
386  delete p;
387  }
388 
389  return 1;
390 }

392 void MDNSResponseScheduler::post(std::shared_ptr<DNSRecord> r, int flush_cache,
393  IPvXAddress* querier, int immediately) {
394  std::shared_ptr<MDNSResponseJob> rj;
395  simtime_t tv;
396  std::string stime = std::to_string(MDNS_RESPONSE_WAIT) + std::string("ms");
397 
398  if ((rj = find_suppressed(r, querier)) && rj->r->ttl >= r->ttl / 2
399  && isGoodbye(r) == isGoodbye(rj->r)) {
400  // response exists and is suppressed..
401  return;
402  }
403 
404  if ((rj = find_history(r))) { // response in history?
405  if ((rj->flush_cache || !flush_cache) && rj->r->ttl >= r->ttl / 2
406  && isGoodbye(r) == isGoodbye(rj->r)) {
407  return;
408  }
409 
410  remove_job(rj); // outdated
411  }
412 
413  if ((rj = find_job(r))) {
414  tv = simTime() + STR_SIMTIME(stime.c_str());
415 
416  if (tv < rj->delivery) {
417  rj->delivery = tv;
418  timeEventSet->updateTimeEvent(rj->e, tv);
419  }
420 
421  rj->flush_cache = flush_cache;
422  rj->querier = querier;
423 
424  freeDnsRecord(rj->r);
425  rj->r = copyDnsRecord(r);
426  return;
427 
428  } else {
429  tv = simTime() + STR_SIMTIME(stime.c_str());
430 
431  rj = new_job(r, 0, 0);
432  rj->delivery = tv;
433  rj->flush_cache = flush_cache;
434 
435  TimeEvent* e = new TimeEvent(this);
436  e->setData(rj);
437  e->setExpiry(tv);
438  e->setLastRun(0);
439  e->setCallback(MDNSResponseScheduler::elapseCallback);
440 
441  rj->e = e;
442  timeEventSet->addTimeEvent(e);
443  if (querier) {
444  rj->querier = querier;
445  }
446  }
447 
448  std::shared_ptr<simtime_t> tv_ptr(new SimTime(tv));
449 
450  callback(tv_ptr, resolver);
451 
452 }

504 void MDNSResponseScheduler::elapse(TimeEvent* e, std::shared_ptr<void> data) {
505  std::shared_ptr<MDNSResponseJob> rj = std::static_pointer_cast
506  < MDNSResponseJob > (data);
507  int packetSize = 12; // initial header size
508  int ancount = 0;
509  std::list<std::shared_ptr<DNSRecord>> anlist;
510  int is_private = 0;
511 
512  if (rj->done || rj->suppressed) {
513  remove_job(rj);
514  return;
515  }
516 
517  std::string service_type;
518  if (hasPrivacy) {
519  // check whether the record is of private nature
520  service_type = extract_stype(rj->r->rname);
521  if (private_service_table->find(service_type)
522  != private_service_table->end()) {
523  std::shared_ptr<PrivateMDNSService> psrv =
524  (*private_service_table)[service_type];
525  is_private = psrv->is_private;
526  }
527  }
528 
529  int success = appendRecord(rj->r, &anlist, &packetSize, &ancount);
530  done(rj);
531 
532  std::list<std::shared_ptr<MDNSResponseJob>> done_records;
533 
534  if (!is_private) {
535  for (auto job : jobs) {
536  if(!success) break;
537 
538  int _private_job = 0;
539  service_type = extract_stype(job->r->rname);
540 
541  // check whether this service is private, do not append it if it is
542  if (hasPrivacy
543  && private_service_table->find(service_type)
544  != private_service_table->end()) {
545  std::shared_ptr<PrivateMDNSService> psrv =
546  (*private_service_table)[service_type];
547  _private_job = psrv->is_private;
548  // reschedule
549  timeEventSet->updateTimeEvent(job->e,
550  simTime() + STR_SIMTIME("20ms"));
551  }
552 
553  if (!_private_job) {
554  success = appendRecord(job->r, &anlist, &packetSize, &ancount);
555  }
556 
557  if (success && !_private_job) {
558  done_records.push_back(job);
559  }
560  }
561  } else {
562  // append records with matching service_type!
563  for (auto job : jobs) {
564  if(!success) break;
565 
566  std::string job_stype = extract_stype(job->r->rname);
567  // append if it is not the same job
568  int not_equal = service_type.compare(job_stype)
569  && recordDataEqual(job->r, rj->r);
570  if (not_equal) {
571  success = appendRecord(job->r, &anlist, &packetSize, &ancount);
572  }
573 
574  if (not_equal && success) {
575  done_records.push_back(job);
576  }
577  }
578  }
579 
580  for(auto job : done_records) done(job);
581 
582  if (ancount == 0) {
583  return;
584  }
585 
586  if (!preparePacketAndSend(anlist, ancount, packetSize, is_private)) {
587 
588  }
