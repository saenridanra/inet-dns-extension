69 void MDNSProbeScheduler::done(std::shared_ptr<ODnsExtension::MDNSProbeJob> pj) {
70  pj->done = 1;
71  auto it = std::find(jobs.begin(), jobs.end(), pj);
72  if (it != jobs.end())
73  jobs.erase(it);
74 
75  if (std::find(history.begin(), history.end(), pj) == history.end())
76  return;
77 
78  history.push_back(pj);
79 
80  simtime_t now = simTime();
81  pj->delivery = now;
82 
83  // update the time event
84 
85  // add random deferral value between 20 and 120
86  int defer = intrand(100) + 20;
87  // create simtime value from random deferral value
88  std::string stime = std::to_string(defer) + std::string("ms");
89  simtime_t tv = STR_SIMTIME(stime.c_str());
90 
91  timeEventSet->updateTimeEvent(pj->e, now + tv);
92 }

107 void MDNSProbeScheduler::remove_job(std::shared_ptr<ODnsExtension::MDNSProbeJob> pj) {
108  timeEventSet->removeTimeEvent(pj->e);
109  auto it = std::find(jobs.begin(), jobs.end(), pj);
110  if (it != jobs.end()) {
111  jobs.erase(it);
112  freeDnsRecord(pj->r);

115 
116  it = std::find(history.begin(), history.end(), pj);
117  if (it != history.end()) {

122

124  freeDnsRecord(pj->r);
125  pj.reset();

128 int MDNSProbeScheduler::preparePacketAndSend(std::list<std::shared_ptr<DNSQuestion>> qlist,
129  std::list<std::shared_ptr<DNSRecord>> nslist, int qdcount, int nscount, int packetSize,
130  int TC, int is_private) {
131  int i = 0;
132  std::string msgname;
133  if (!is_private)
134  msgname = "MDNS_probe#";
135  else
136  msgname = "PRIVATE_probe#";
137 
138  msgname += std::to_string(id_count);
139 
140  DNSPacket* p = ODnsExtension::createNQuery(msgname, qdcount, 0, nscount, 0,
141  id_count++, 0);
142 
143  // append questions
144  for (auto it : qlist) {
145  ODnsExtension::appendQuestion(p, it, i);
146  i++;
147  }
148 
149  // append auth if available
150  if (nscount > 0) {
151  i = 0;
152  for (auto it : nslist) {
153  ODnsExtension::appendAuthority(p, it, i);
154  i++;
155  }
156  }
157 
158  // packet fully initialized, send it via multicast
159  p->setByteLength(packetSize);
160  if (!is_private) {
161  const char* dstr = "i=msg/bcast,red";
162  p->setDisplayString(dstr);
163  outSock->sendTo(p, multicast_address, MDNS_PORT);
164  } else {
165  const char* dstr = "i=msg/packet,green";
166  p->setDisplayString(dstr);
167  std::string service_type = ODnsExtension::extract_stype(
168  p->getQuestions(0).qname);
169  std::shared_ptr<ODnsExtension::PrivateMDNSService> psrv =
170  (*private_service_table)[service_type];
171  // go through the offered_to list
172 
173  for (auto it : psrv->offered_to) {
174  std::string key = it;
175  std::shared_ptr<ODnsExtension::FriendData> fdata = (*friend_data_table)[key];
176  if (fdata && fdata->online) {
177  // send per TCP to the privacy socket on the given port
178  privacySock->sendTo(p->dup(), fdata->address, fdata->port);
179  }
180  }
181 
182  delete p;
183  }
184 
185  // packet is out, we're finished
186  return 1;
187 }

189 int MDNSProbeScheduler::append_question(std::shared_ptr<ODnsExtension::MDNSProbeJob> pj,
190  std::list<std::shared_ptr<DNSQuestion>>* qlist, std::list<std::shared_ptr<DNSRecord>>* nslist,
191  int *packetSize, int* qdcount, int* nscount, int is_private) {
192  std::shared_ptr<ODnsExtension::DNSQuestion> q;
193 
194  int pack_has_space = 1;
195 
196  // ANY question for probing
197  q = createQuestion(pj->r->rname, DNS_TYPE_VALUE_ANY, DNS_CLASS_IN);
198 
199  int qsize = sizeof(pj->r->rname) + 4; // name length + 4 bytes for type and class
200 
201  if (*packetSize + qsize > MAX_MDNS_PACKET_SIZE) {
202  return 0;
203  }
204 
205  // check if record fits
206  int size = 10 + sizeof(pj->r->rname) + pj->r->rdlength;
207 
208  if (*packetSize + size > MAX_MDNS_PACKET_SIZE) {
209  return 0;
210  }
211 
212  *packetSize += qsize + size;
213 
214  (*qdcount)++;
215  (*qlist).push_back(q);
216  (*nscount)++;
217  (*nslist).push_back(ODnsExtension::copyDnsRecord(pj->r));
218 
219  done(pj);
220 
221  // now see if there are more records that match..
222 
223  std::list<std::shared_ptr<MDNSProbeJob>> done_records;
224  for (auto job : jobs) {
225  // check if key matches ..
226  if (!ODnsExtension::recordEqualNoData(job->r, pj->r)) {
227  // record does not match...
228  continue;
229  }
230 
231  // check if record fits
232  size = 10 + sizeof(job->r->rname) + job->r->rdlength;
233 
234  if (*packetSize + size > MAX_MDNS_PACKET_SIZE) {
235  pack_has_space = 0;
236  break;
237  }
238 
239  *packetSize += size;
240 
241  // append record
242  (*nscount)++;
243  (*nslist).push_back(ODnsExtension::copyDnsRecord(job->r));
244 
245  done_records.push_back(job);
246  }
247 
248  // mark all PJs in the list as done
249  for (auto it : done_records)
250  done(it);
251 
252  return pack_has_space;
253 }

261 void MDNSProbeScheduler::post(std::shared_ptr<ODnsExtension::DNSRecord> r, int immediately) {
262  std::shared_ptr<MDNSProbeJob> pj;
263  simtime_t tv;
264 
265  if ((pj = find_history(r)))
266  return; // still got a record in the history for this probe
267 
268  if (!immediately) {
269  int defer = MDNS_PROBE_WAIT;
270  // create simtime value from random deferral value
271  std::string stime = std::to_string(defer) + std::string("ms");
272  tv = simTime() + STR_SIMTIME(stime.c_str());
273 
274  } else {
275  tv = simTime();
276  }
277 
278  if ((pj = find_job(r))) {
279  if (tv < pj->delivery) {
280  pj->delivery = tv;
281  timeEventSet->updateTimeEvent(pj->e, tv);
282  }
283  } else {
284  // create new job..
285  pj = new_job(r);
286  pj->delivery = tv;
287 
288  ODnsExtension::TimeEvent* e = new ODnsExtension::TimeEvent(this);
289  e->setData(pj);
290  e->setExpiry(tv);
291  e->setLastRun(0);
292  e->setCallback(ODnsExtension::MDNSProbeScheduler::elapseCallback);
293 
294  pj->e = e;
295  timeEventSet->addTimeEvent(e);
296  }
297 
298  std::shared_ptr<SimTime> tv_ptr(new SimTime(tv));
299 
300  callback(tv_ptr, resolver);
301 }

303 void MDNSProbeScheduler::elapse(ODnsExtension::TimeEvent* e, std::shared_ptr<void> data) {
304  // elapse callback, cast probejob
305  std::shared_ptr<MDNSProbeJob> pj = std::static_pointer_cast<MDNSProbeJob>(data);
306  int is_private = 0;
307  std::shared_ptr<ODnsExtension::PrivateMDNSService> psrv;
308 
309  if (pj->done) {
310  remove_job(pj);
311  return;
312  }
313 
314  if (hasPrivacy) {
315  // check whether the record is of private nature
316  std::string service_type = ODnsExtension::extract_stype(pj->r->rname);
317  if (private_service_table->find(service_type) != private_service_table->end()) {
318  psrv = (*private_service_table)[service_type];
319  is_private = psrv->is_private;
320  }
321  }
322 
323  int packetSize = 12; // initial header size
324 
325  std::list<std::shared_ptr<DNSQuestion>> qlist;
326  std::list<std::shared_ptr<DNSRecord>> nslist;
327 
328  int qdcount = 0;
329  int nscount = 0;
330 
331  int success = append_question(pj, &qlist, &nslist, &packetSize, &qdcount,
332  &nscount, is_private);
333 
334  // now try to append more questions if we didn't already exceed the packet size.
335  // take another query job from the list..
336 
337  std::list<std::shared_ptr<MDNSProbeJob>> list_cpy;
338  list_cpy.insert(list_cpy.end(), jobs.begin(), jobs.end());
339  if (!is_private) { // only do so if the job was not private, we already appended all matching keys
340  for (auto job : list_cpy) {
341  if(!success) break;
342 
343  int _private_job = 0;
344  std::string service_type = ODnsExtension::extract_stype(job->r->rname);
345 
346  // check whether this service is private, do not append it if it is
347  if (hasPrivacy && private_service_table->find(service_type) != private_service_table->end()) {
348  psrv = (*private_service_table)[service_type];
349  _private_job = psrv->is_private;
350  // reschedule
351  timeEventSet->updateTimeEvent(job->e,
352  simTime() + STR_SIMTIME("20ms"));
353 
354  }
355 
356  if (!_private_job) {
357  success = append_question(job, &qlist, &nslist, &packetSize,
358  &qdcount, &nscount, is_private);
359  }
360 
361  if (success && !_private_job) {
362  done(job);
363  }
364  }
365  }
366 
367  if (preparePacketAndSend(qlist, nslist, qdcount, nscount, packetSize,
368  !success, is_private)) {
369  // success, delegate?
370  } else {
371  // some error message?
372  }
373 
374 }
