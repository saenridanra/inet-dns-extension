139 void MDNSAnnouncer::add_service(std::shared_ptr<MDNSService> service) {
140  // Create an SRV record for the service
141  std::shared_ptr<DNSRecord> service_record(new DNSRecord());
142  std::string label = service->name + std::string(".")
143  + service->service_type;
144 
145  service_record->rname = label;
146  service_record->rtype = DNS_TYPE_VALUE_SRV;
147  service_record->rclass = DNS_CLASS_IN;
148  service_record->ttl = 60 * 75;
149 
150  std::shared_ptr<SRVData> srv(new SRVData());
151  srv->name = label;
152  srv->target = target;
153  srv->service = service->service_type;
154  srv->port = service->port;
155  srv->ttl = 60 * 75;
156  srv->weight = 0;
157  srv->priority = 0;
158  srv->proto = "_tcp"; // for now just stub tcp in..
159 
160  service_record->rdata = srv;
161  service_record->rdlength = 6 + label.length() + target.length()
162  + service->service_type.length();
163 
164  std::shared_ptr<Probe> p(new Probe());
165  p->r = service_record;
166  p->probe_id = id_internal++;
167  p->ref_service = service;
168 
169  probing.push_back(p);
170 
171  simtime_t tv = simTime() + STR_SIMTIME("20ms");
172 
173  ODnsExtension::TimeEvent* e = new ODnsExtension::TimeEvent(this);
174  e->setData(p);
175  e->setExpiry(tv);
176  e->setLastRun(0);
177  e->setCallback(ODnsExtension::MDNSAnnouncer::elapseCallback);
178  p->e = e;
179  timeEventSet->addTimeEvent(e);
180 
181  // go through all txt records
182  if (service->txtrecords.size() > 0) {
183  for (auto it : service->txtrecords) {
184  std::shared_ptr<DNSRecord> txtrecord(new DNSRecord());
185  txtrecord->rname = label;
186  txtrecord->rtype = DNS_TYPE_VALUE_TXT;
187  txtrecord->rclass = DNS_CLASS_IN;
188  txtrecord->strdata = it;
189  txtrecord->rdlength = it.length();
190  txtrecord->ttl = MDNS_SERVICE_TTL;
191 
192  p = std::shared_ptr < Probe > (new Probe());
193  p->r = txtrecord;
194  p->probe_id = id_internal++;
195  p->ref_service = service;
196 
197  e = new ODnsExtension::TimeEvent(this);
198  e->setData(p);
199  e->setExpiry(tv);
200  e->setLastRun(0);
201  e->setCallback(ODnsExtension::MDNSAnnouncer::elapseCallback);
202  timeEventSet->addTimeEvent(e);
203 
204  p->e = e;
205  }
206  } else {
207  // add an empty txt record..
208  std::shared_ptr<DNSRecord> txtrecord(new DNSRecord());
209  std::string txt = "";
210  txtrecord->rname = label;
211  txtrecord->rtype = DNS_TYPE_VALUE_TXT;
212  txtrecord->rclass = DNS_CLASS_IN;
213  txtrecord->strdata = txt;
214  txtrecord->rdlength = 0;
215  txtrecord->ttl = MDNS_SERVICE_TTL;
216 
217  p = std::shared_ptr < Probe > (new Probe());

247 int MDNSAnnouncer::check_conflict(std::shared_ptr<DNSRecord> r) {
248  // check if probe list is non-empty
249  int conflict = 0;
250 
251  for (auto p : probing) {
252  if (ODnsExtension::recordDataEqual(r, p->r)) {
253  conflict = 1;
254  break; // the other host announces the exact same record, we're finished here
255  } else if (ODnsExtension::recordEqualNoData(r, p->r)
256  && p->s == ProbeState::PROBING) {
257  // we have a conflict, remove the probe, and try with another label
258  // since there may be more probes matching this label, we withdraw all of them
259 
260  if (r->rtype == DNS_TYPE_VALUE_SRV && p->r->rdata != r->rdata) {
261  withdraw(p);
262  conflict = 1;
263  } else if (p->r->strdata != r->strdata) {
264  withdraw(p);
265  conflict = 1;
266  }
267  } else {
268  // we're already announcing the probe, it belongs to us
269  break;
270  }
271  }
272 
273  return conflict;
274 }

276 void MDNSAnnouncer::withdraw(std::shared_ptr<Probe> p) {
277  // by withdrawing, the label is changed and the probing number reset
278  p->n_iter = 0;
279  std::string label_new = "";
280 
281  if (p->ref_service) {
282  label_new = p->ref_service->name + std::string("-")
283  + std::to_string(++p->collision_count) + std::string(".")
284  + p->ref_service->service_type;
285  }
286 
287  // use the new label..
288  p->s = ProbeState::PROBING;
289  p->r->rname = label_new;
290 }

292 void MDNSAnnouncer::goodbye(std::shared_ptr<Probe> p, int send_goodbye,
293  int remove) {
294  if (send_goodbye) {
295  if (p->s == ProbeState::ANNOUNCING) {
296  std::shared_ptr<DNSRecord> goodbye_record =
297  ODnsExtension::copyDnsRecord(p->r);
298  goodbye_record->ttl = 0;
299  response_scheduler->post(goodbye_record, 0, NULL, 0);
300  }
301  }
302 
303  if (remove) {
304  // delete the probe and related time events
305  timeEventSet->removeTimeEvent(p->e);
306  freeDnsRecord(p->r);
307  delete p->e;
308  p.reset();
309  }

312 void MDNSAnnouncer::elapse(ODnsExtension::TimeEvent* e,
313  std::shared_ptr<void> data) {
314  std::shared_ptr<Probe> p = std::static_pointer_cast < Probe > (data);
315  simtime_t tv;
316  // no probe has been sent out so far..
317  if (p->s == ProbeState::STARTING) {
318  // do the first probe
319  p->n_iter++;
320  probe_scheduler->post(p->r, 0);
321  p->s = ProbeState::PROBING;
322 
323  // 250msec delay
324  tv = simTime() + STR_SIMTIME("250ms");
325  timeEventSet->updateTimeEvent(p->e, tv);
326 
327  } else if (p->s == ProbeState::PROBING) {
328  if (p->n_iter > 2) {
329  // we're finished probing, start announcing
330  // send first response for probe
331  p->n_iter = 1;
332  p->s = ProbeState::ANNOUNCING;
333  response_scheduler->post(p->r, 1, NULL, 0);
334 
335  // add 2^(p->n_iter - 1) * 1s delay
336  tv = simTime()
337  + ((int) pow(2, (p->n_iter - 1)) * STR_SIMTIME("1s"));
338  timeEventSet->updateTimeEvent(p->e, tv);
339  } else {
340  // still need to send out some probes
341  p->n_iter++;
342  probe_scheduler->post(p->r, 0);
343  // 250msec delay
344  tv = simTime() + STR_SIMTIME("250ms");
345  timeEventSet->updateTimeEvent(p->e, tv);
346  }
347  } else if (p->s == ProbeState::ANNOUNCING) {
348  if (p->n_iter > 3) {
349  if (s == AnnouncerState::START || s == AnnouncerState::RESTART) {
350  s = AnnouncerState::PROBE;
351  initialize();
352  } else if (s == AnnouncerState::PROBE
353  && num_announced_jobs >= probing.size()) {
354  s = AnnouncerState::FINISHED;
355  }
356 
357  if (p->s != ProbeState::ANNOUNCED) {
358  num_announced_jobs++;
359  p->s = ProbeState::ANNOUNCED;
360  p->n_iter = 0;
361  }
362  // add record to cache..
363 
364  auth_cache->put_into_cache(p->r); // using the cache, we know when the record is up for eviction
365  std::string hash = p->r->rname + std::string(":")
366  + std::string(getTypeStringForValue(p->r->rtype))
367  + std::string(":")
368  + std::string(getClassStringForValue(p->r->rclass));
369  probe_to_cache[p->probe_id] = hash;
370 
371  // add 2^(p->n_iter - 1) * 1s delay
372  tv = simTime()
373  + (((int) pow(2, (p->n_iter - 1)) * STR_SIMTIME("1s")));
374  timeEventSet->updateTimeEvent(p->e, tv);
375  } else {
376  // still have to send out annoucements
377  p->n_iter++;
378  response_scheduler->post(p->r, 1, NULL, 0);
379 
380  // add 2^(p->n_iter - 1) * 1s delay
381  tv = simTime()
382  + (((int) pow(2, (p->n_iter - 1)) * STR_SIMTIME("1s")));
383  timeEventSet->updateTimeEvent(p->e, tv);
384  }
385  } else if (p->s == ProbeState::ANNOUNCED) {
386  // don't put into cache, but send a response every 2^n s with max 60 min.
387  p->n_iter++;
388  response_scheduler->post(p->r, 1, NULL, 0);
389 
390  // add 2^(p->n_iter - 1) * 1s delay
391  int scnds = ((int) pow(2, (p->n_iter - 1)));
392  if (scnds > 3600)
393  tv = simTime() + (3600 * STR_SIMTIME("1s"));
394  else
395  tv = simTime() + (scnds * STR_SIMTIME("1s"));
396  timeEventSet->updateTimeEvent(p->e, tv);
397  }
398 }
