87

93  else

95

97  + std::string(":") + std::string(DNS_CLASS_STR_IN);
98 
99  has_ns_reference = config->hasEntry(ns_reference_hash);
100 
101  if (!has_ns_reference) {
102  // ok we did not find a reference directly, do a longest suffix match
103  // to check if there is a suffix matching
104  std::unordered_map<std::string,
105  std::list<std::shared_ptr<zone_entry>>>* zone =
106  config->getEntries();
107  unsigned int max_len = 0;
108  std::string tmp_ref_hash = "";
109  for (auto it = (*zone).begin(); it != (*zone).end(); ++it) {
110  if (ODnsExtension::stdstr_has_suffix(ns_reference_hash,
111  it->first)) {
112  if (it->first.length() > max_len) {
113  max_len = it->first.length();
114  tmp_ref_hash = std::string(it->first);
115  }
116  }
117  }
118 
119  if (tmp_ref_hash != "") {
120  ns_reference_hash = std::string(tmp_ref_hash);
121  has_ns_reference = 1;
122  }
123 
124  }
125  }
126 
127  // generate msg name
128  std::string msg_name = std::string("dns_response#")
129  + std::to_string(response_count++);
130 
131  if (pos > 0 && config->getOrigin() != "." && !has_ns_reference) {
132  is_authoritative = 1;
133  }
134 
135  // init class
136  switch (q.qclass) {
137  case DNS_CLASS_IN:

139  break;
140  case DNS_CLASS_CH:
141  case DNS_CLASS_CS:
142  case DNS_CLASS_HS:

154  case DNS_TYPE_VALUE_AAAA:

156  break;
157  case DNS_TYPE_VALUE_CNAME:
158  type = DNS_TYPE_STR_CNAME;
159  break;
160  case DNS_TYPE_VALUE_HINFO:
161  response = DNSServerBase::unsupportedOperation(query);
162  return response;
163  case DNS_TYPE_VALUE_MINFO:
164  type = DNS_TYPE_STR_MINFO;
165  break;
166  case DNS_TYPE_VALUE_MX:
167  type = DNS_TYPE_STR_MX;
168  break;
169  case DNS_TYPE_VALUE_NS:
170  type = DNS_TYPE_STR_NS;
171  break;
172  case DNS_TYPE_VALUE_NULL:
173  response = DNSServerBase::unsupportedOperation(query);
174  return response;
175  case DNS_TYPE_VALUE_PTR:
176  type = DNS_TYPE_STR_PTR;
177  break;
178  case DNS_TYPE_VALUE_SOA:
179  type = DNS_TYPE_STR_SOA;
180  break;
181  case DNS_TYPE_VALUE_TXT:
182  type = DNS_TYPE_STR_TXT;
183  break;
184  case DNS_TYPE_VALUE_SRV:
185  type = DNS_TYPE_STR_SRV;
186  break;
187  case DNS_TYPE_VALUE_ANY:
188  type = DNS_TYPE_STR_ANY;
189  break;
190  case DNS_TYPE_VALUE_AXFR:
191  type = DNS_TYPE_STR_AXFR;
192  break;
193  }
194 
195  // now we can lookup the name in the database
196  if (is_authoritative) {
197  if (q.qtype == DNS_TYPE_VALUE_ANY) {
198  // or the type..
199  const char** type_array = ODnsExtension::getTypeArray();
200  for (uint32_t i = 0; i < sizeof(type_array); i++) {
201  std::string type_str = std::string(type_array[i]);
202  namehash = q.qname;
203  if (ODnsExtension::stdstr_has_suffix(q.qname, std::string(".")))
204  namehash = namehash + std::string(":");
205  else
206  namehash = namehash + std::string(".:");
207 
208  namehash = namehash + std::string(":") + std::string(__class);
209 
210  // we basically just have to append every record that matches
211  // the hash to the answer section
212  answer_list = appendEntries(namehash, answer_list,
213  ODnsExtension::getTypeValueForString(type_str),
214  &an_records);
215  }
216 
217  } else {
218  // we have a specific record, so lets look it up in the hash table
219  namehash = q.qname;
220  if (ODnsExtension::stdstr_has_suffix(q.qname, std::string(".")))
221  namehash = namehash + std::string(":");
222  else
223  namehash = namehash + std::string(".:");
224 
225  namehash = namehash + std::string(type) + std::string(":")
226  + std::string(__class);
227

229  &an_records);
230 
231  if (an_records > 0) {
232 
233  // fill out authority section with NS records
234  // if the request was on the ZONE ORIGIN
235  // and it was not made for NS records.
236 
237  if (q.qtype != DNS_TYPE_VALUE_A
238  && q.qtype != DNS_TYPE_VALUE_AAAA) {
239  ar_list = appendTransitiveEntries(answer_list, ar_list,
240  DNS_TYPE_STR_A, DNS_TYPE_VALUE_A, &ar_records);
241  ar_list = appendTransitiveEntries(answer_list, ar_list,
242  DNS_TYPE_STR_AAAA, DNS_TYPE_VALUE_AAAA, &ar_records);
243  }
244 
245  // add CNAME if there are links for this label
246  if (q.qtype != DNS_TYPE_VALUE_CNAME) {
247  // get NS records
248  cnhash = q.qname;
249  if (ODnsExtension::stdstr_has_suffix(q.qname,
250  std::string(".")))
251  cnhash = q.qname + std::string(":");
252  else
253  cnhash = q.qname + std::string(".:");
254 
255  cnhash = cnhash + std::string(DNS_TYPE_STR_CNAME)
256  + std::string(":") + std::string(__class);
257 
258  answer_list = appendEntries(cnhash, answer_list,
259  DNS_TYPE_VALUE_CNAME, &an_records);
260 
261  // now we don't add the QTYPE record to additional, but to the answer section
262  // iterate through the CNAME records, add the QTYPE records accordingly....
263  for (auto it = answer_list.begin(); it != answer_list.end();
264  ++it) {
265  std::shared_ptr<DNSRecord> t_r = *it;
266  // use the data string to create a hash and find the qtype record
267  std::string namecpy = std::string(t_r->strdata);
268  cnhash = namecpy + std::string(".")
269  + config->getOrigin() + std::string(":")
270  + std::string(type) + std::string(":")
271  + std::string(__class);
272 
273  // get entries for type using this hash
274  answer_list = appendEntries(cnhash, answer_list,
275  q.qtype, &an_records);
276  }
277  }
278  } else {
279  // add CNAME if there are links for this label
280  if (q.qtype != DNS_TYPE_VALUE_CNAME) {
281  // get NS records
282  cnhash = q.qname;
283  if (ODnsExtension::stdstr_has_suffix(q.qname,
284  std::string(".")))
285  cnhash = q.qname + std::string(":");
286  else
287  cnhash = q.qname + std::string(".:");
288 
289  cnhash = cnhash + std::string(DNS_TYPE_STR_CNAME)
290  + std::string(":") + std::string(__class);
291 
292  answer_list = appendEntries(cnhash, answer_list,
293  DNS_TYPE_VALUE_CNAME, &an_records);
294 
295  for (auto it = answer_list.begin(); it != answer_list.end();
296  ++it) {
297  std::shared_ptr<DNSRecord> t_r = *it;
298  // use the data string to create a hash and find the qtype record
299  std::string namecpy = std::string(t_r->strdata);
300  cnhash = namecpy + std::string(".")
301  + config->getOrigin() + std::string(":")
302  + std::string(type) + std::string(":")
303  + std::string(__class);
304 
305  // get entries for type using this hash
306  answer_list = appendEntries(cnhash, answer_list,
307  q.qtype, &an_records);
308  }
309  }
310  }
311 
312  }
313 
314  if (an_records == 0) { // no entry found, although authoritative
315  // append SOA
316  response = ODnsExtension::createResponse(msg_name, 1, 0, 0, 0, id,
317  opcode, 1, rd, ra, 3);
318  } else {
319  // append authority
320  ns_list = appendAuthority(ns_list, &ns_records);
321  if (ns_records > 0) {
322  ar_list = appendAdditionals(ns_list, ar_list, &ar_records);
323  }
324 
325  response = ODnsExtension::createResponse(msg_name, 1, an_records,
326  ns_records, ar_records, id, opcode, 1, rd, ra, 0);
327  }
328 
329  ODnsExtension::appendQuestion(response,
330  ODnsExtension::copyDnsQuestion(&q), 0);
331  } else {
332  // we're not authoritative, but maybe we have an entry that points
333  // to an NS that is authoritative for a suffix of the query
334 
335  bool found_entry = false;
336 
337  // tokenize the question label. check if theres an entry for the last suffix, i.e. last token
338  std::vector<std::string> tokens =
339  cStringTokenizer(q.qname.c_str(), ".").asVector();
340  // check last token and create hash using NS
341  std::string reference_hash = tokens[tokens.size() - 1]
342  + std::string(".:") + std::string(DNS_TYPE_STR_NS)
343  + std::string(":") + std::string(__class);
344 
345  if (has_ns_reference) { // see if we know a nameserver on the prefix
346  found_entry = true;
347  ns_list = appendEntries(ns_reference_hash, ns_list,
348  DNS_TYPE_VALUE_NS, &ns_records);
349 
350  ar_list = appendTransitiveEntries(ns_list, ar_list, DNS_TYPE_STR_A,
351  DNS_TYPE_VALUE_A, &ar_records);
352  ar_list = appendTransitiveEntries(ns_list, ar_list,
353  DNS_TYPE_STR_AAAA, DNS_TYPE_VALUE_AAAA, &ar_records);
354 
355  // response with with no AA set
356  response = ODnsExtension::createResponse(msg_name, 1, an_records,
357  ns_records, ar_records, id, opcode, 0, rd, ra, 0);
358 
359  // set question
360  ODnsExtension::appendQuestion(response,
361  ODnsExtension::copyDnsQuestion(&q), 0);
362 
363  } else if (config->hasEntry(reference_hash)) {
364  found_entry = true;
365  ns_list = appendEntries(reference_hash, ns_list, DNS_TYPE_VALUE_NS,
366  &ns_records);
367 
368  if (ns_records != 0) {
369 
370  ar_list = appendTransitiveEntries(ns_list, ar_list,
371  DNS_TYPE_STR_A, DNS_TYPE_VALUE_A, &ar_records);
372  ar_list = appendTransitiveEntries(ns_list, ar_list,
373  DNS_TYPE_STR_AAAA, DNS_TYPE_VALUE_AAAA, &ar_records);
374 
375  // response with with no AA set
376  response = ODnsExtension::createResponse(msg_name, 1,
377  an_records, ns_records, ar_records, id, opcode, 0, rd,
378  ra, 0);
379  }
380  }
381 
382  if (!found_entry) {
383 
384  if (recursion_available && rootServers.size() > 0) {
385  if (rd) {
386  // assign an id for the query cache
387  int id = DNSServerBase::getIdAndInc();
388  DNSServerBase::store_in_query_cache(id, query);
389  msg_name = std::string("dns_query#") + std::to_string(id)
390  + std::string("--recursive");
391 
392  // do the initial query towards a root server
393  // pick at random
394  int p = intrand(rootServers.size());
395  DNSPacket *root_q = ODnsExtension::createQuery(msg_name,
396  query->questions[0].qname, DNS_CLASS_IN,
397  query->questions[0].qtype, id, 1);
398 
399  DNSServerBase::sendResponse(root_q, rootServers[p]);
400 
401  return NULL; // so it is known that recursive resolving has been initiated
402 
403  } else {
404  // response with not found err
405  response = ODnsExtension::createResponse(msg_name, 1,
406  an_records, ns_records, ar_records, id, opcode, 0,
407  rd, ra, 0);
408  }
409 
410  // set question
411  ODnsExtension::appendQuestion(response,
412  ODnsExtension::copyDnsQuestion(&q), 0);
413  } else {
414  //TODO; Iterative answer.
415  response = DNSServerBase::unsupportedOperation(query);
416  return response;
417  }
418  }
419  }
420 
421  // append an, ns, ar to response
422  int index = 0;
423  if (an_records > 0) {
424  for (auto it = answer_list.begin(); it != answer_list.end(); ++it) {
425  ODnsExtension::appendAnswer(response, *it, index++);
426  }
427  }
428  if (ns_records > 0) {
429  index = 0;
430  for (auto it = ns_list.begin(); it != ns_list.end(); ++it) {
431  ODnsExtension::appendAuthority(response, *it, index++);
432  }
433  }
434  if (ar_records > 0) {
435  index = 0;
436  for (auto it = ar_list.begin(); it != ar_list.end(); ++it) {
437  ODnsExtension::appendAdditional(response, *it, index++);
438  }
439  }
440  if (an_records == 0 && ns_records == 0 && ar_records == 0) {
441  // just append the SOA entry.
442  }
443 
444  return response;
445

468 std::list<std::shared_ptr<DNSRecord>> DNSAuthServer::appendEntries(
469  std::string hash, std::list<std::shared_ptr<DNSRecord>> dstlist,
470  int type, int *num_records) {
471  std::list<std::shared_ptr<zone_entry>> entries = config->getEntry(hash);
472  std::shared_ptr<ODnsExtension::DNSRecord> rr;
473 
474  for (auto entry : entries) {
475  rr = std::shared_ptr < ODnsExtension::DNSRecord > (new DNSRecord());
476  rr->rdata = NULL;
477  rr->strdata = std::string(entry->data);
478 
479  if (ODnsExtension::stdstr_has_suffix(entry->domain,
480  config->getOrigin())) {
481  rr->rname = std::string(entry->domain);
482  } else {
483  rr->rname = std::string(entry->domain) + std::string(".")
484  + std::string(config->getOrigin());
485  }
486 
487  rr->rclass = (short) DNS_CLASS_IN;
488  rr->rtype = (short) type;
489  rr->rdlength = entry->data.length();
490  rr->ttl = config->getTTL();
491 
492  dstlist.push_back(rr);
493  (*num_records)++;
494  }
495  return dstlist;
496 }
