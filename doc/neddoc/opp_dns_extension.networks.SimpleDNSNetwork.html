network SimpleDNSNetwork
{
    @display("bgb=1071,606");
    submodules:
        dns1: DNSServerNode {
            parameters:
                m_file = "pan.rz.uni-konstanz.de.zone";
                @display("p=532,303");
        }
        cli1: DNSClientNode {
            @display("p=403,303");
        }
        configurator: IPv4NetworkConfigurator {
            @display("p=121,37");
        }

        router: Router {
            @display("p=477,206");
        }
    connections:
        cli1.ethg++ <--> Eth1G <--> router.ethg++;
        dns1.ethg++ <--> Eth1G <--> router.ethg++;
}
