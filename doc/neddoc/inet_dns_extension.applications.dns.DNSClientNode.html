module DNSClientNode extends StandardHost
{
    parameters:
        hasUdp = true;

    submodules:
        cli: DNSClientTraffGen;
    connections:
        cli.udpOut --> udp.appIn++;
        cli.udpIn <-- udp.appOut++;
}
