    unsigned short options;

                                @sizeSetter(setNumAuthorities)
                                @sizetype(short);
    
    DNSRecord   additional[]    @sizeGetter(getNumAdditional)
                                @sizeSetter(setNumAdditional)
                                @sizetype(short);
}
